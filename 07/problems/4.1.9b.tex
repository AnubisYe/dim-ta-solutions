\problem{9}
In each case, find $\overrightarrow{PQ}$ and $\|\overrightarrow{PQ}\|$.
\begin{enumerate}[a)]
	\setcounter{enumi}{1}
	\item $P(2, 0, 1)$, $Q(1, -1, 6)$
\end{enumerate}

\solution
\begin{enumerate}[a)]
	\setcounter{enumi}{1}
	\item 
		\begin{align*}
			\overrightarrow{PQ}
			=
			\begin{bmatrix}
				1 - 2 \\
				(-1) - 0 \\
				6 - 1
			\end{bmatrix}
			=
			\begin{bmatrix*}[r]
				-1 \\
				-1 \\
				5
			\end{bmatrix*}
			&,&
			\|\overrightarrow{PQ}\|
			=
			\sqrt{(-1)^2 + (-1)^2 + 5^2}
			=
			\sqrt{27}
		\end{align*}
\end{enumerate}
