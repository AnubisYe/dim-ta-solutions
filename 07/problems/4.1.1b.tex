\problem{1}
Compute $\|\vec{v}\|$ if $\vec{v}$ equals:
\begin{enumerate}[a)]
	\setcounter{enumi}{1}
	\item $\begin{bmatrix}1&-1&2\end{bmatrix}^T$
\end{enumerate}

\solution
\begin{enumerate}[a)]
	\setcounter{enumi}{1}
	\item $\sqrt{1^2 + (-1)^2 + 2^2} = \sqrt{1 + 1 + 4} = \sqrt{6}$
\end{enumerate}
