\problem{2}
Find the inverse of each of the following matrices.
\begin{enumerate}[a)]
	\setcounter{enumi}{1}
	\item 
		$\left[\begin{array}{rr}
			4 & 1 \\
			3 & 2
		\end{array}\right]$
	\setcounter{enumi}{3}
	\item 
		$\left[\begin{array}{rrr}
			1 & -1 & 2 \\
			-5 & 7 & -11 \\
			-2 & 3 & -5
		\end{array}\right]$
\end{enumerate}


\solution

\begin{enumerate}[a)]
    \item
        Set up a matrix with our matrix on the left and the identity on the right. 
		$\left[\begin{array}{rr|rr}
			4 & 1 & 1 & 0 \\
			3 & 2 & 0 & 1
		\end{array}\right]$
        Use Gauss to get the identity on the left side and what remains on the right side is our inverse. 
		$\left[\begin{array}{rr|rr}
			1 & 0 & \frac{2}{5} & \frac{-1}{5} \\
			0 & 1 & \frac{-3}{5} & \frac{4}{5}
		\end{array}\right]$
    \setcounter{enumi}{3}
	\item 
        Set up a matrix with our matrix on the left and the identity on the right.
		$\left[\begin{array}{rrr|rrr}
			1 & -1 & 2 & 1 & 0 & 0 \\
			-5 & 7 & -11 & 0 & 1 & 0\\
			-2 & 3 & -5 & 0 & 0 & 1
		\end{array}\right]$
        Use Gauss to get the identity on the left side and what remains on the right side is our inverse.
		$\left[\begin{array}{rrr|rrr}
			1 & 0 & 0 & 2 & -1 & 3 \\
			0 & 1 & 0 & 3 & 1 & -1\\
			0 & 0 & 1 & 1 & 1 & -2
		\end{array}\right]$
\end{enumerate}
