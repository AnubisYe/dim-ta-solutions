\problem{4}
Given
$A^{-1} = 
\left[\begin{array}{rrr}
	1 & -1 & 3 \\
	2 & 0 & 5 \\
	-1 & 1 & 0
\end{array}\right]$
:
\begin{enumerate}[a)]
	\item Solve the system of equations
		$
			AX =
			\left[\begin{array}{rrr}
				1 \\
				-1 \\
				3
			\end{array}\right]
		$.
\end{enumerate}
