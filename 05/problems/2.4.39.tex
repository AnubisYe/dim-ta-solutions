\problem{39}
	Find $\sum_{k=100}^{200} k$. (Use Table 2.)

\solution
	\begin{align*}
		\sum_{k=100}^{200} k &= \sum_{k=1}^{200} k - \sum_{k=1}^{99} k  \\
		                     &= \frac{200(200 + 1)}{2} - \frac{99(99 + 1)}{2} \\
		                     &= \frac{40200}{2} - \frac{9900}{2} \\
		                     &= \frac{40200 - 9900}{2} \\
		                     &= \frac{30300}{2} \\
		                     &= 15150
	\end{align*}
