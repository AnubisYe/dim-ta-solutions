\problem{1}
Let $A = \{0, 11\}$ and $B = \{00, 01\}$.
Find each of these sets.
\begin{enumerate}[a)]
	\item $AB$
	\item $BA$
	\item $A^2$
	\item $B^3$
\end{enumerate}

\solution
\begin{enumerate}[a)]
	\item $\{000, 001, 1100, 1101\}$
	\item $\{000, 0011, 010, 0111\}$
	\item $\{00, 011, 110, 1111\}$
	\item $\{000000, 000001, 000100, 010000, 000101, 010001, 010100, 010101\}$
\end{enumerate}
